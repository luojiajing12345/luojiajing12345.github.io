function tag(tagName){

创建一个生成标签，并给元素类名（row）的函数
function tag(tagName,attributes){
var element = document.createElement(tagName)
for(var key in attributes){
element[key] = attributes[key]
}
return element

}
divxxx = tag('div',{className:'row'})//创建一个div标签，它的classname是row,把attributes的key，也就是className赋值给element的key。

//attributes是一个属性，我们要遍历这个属性。for…in..循环可以遍历一个哈希。
element也是一个哈希，div.className='xxx',div也是个哈希，className是个key，xxx是key对应的value。
className、id、textContent都是key。
使用函数封装后，代码变成树形的，然后这样更容易理解，更容易区分，更容易排错
